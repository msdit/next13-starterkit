import { Meta } from "@storybook/blocks";
import theme from "../../tailwind.config.ts"

export const { colors } = theme.theme
export const colorList = Object.keys(colors).reduce((res, color) => [ ...res, {
    title: color,
    children: Object.keys(colors[color]).reduce((res, child) => [ ...res, {
        title: child,
        hex: colors[color][child]
    }], [])
}], [])

<Meta title="Design System" />

<div className="container mx-auto">
    # Colors
    {
        colorList.map(el => <div>
            <h2>{el.title}</h2>
            <div className="flex gap-2 flex-wrap pb-8">
            {
                el.children.map(child => (
                    <div style={{background: child.hex}} className="px-8 py-4 rounded shadow">
                        <div className="w-24 py-0 px-2 rounded backdrop-brightness-50 bg-white/30 text-center">
                            {child.title}
                        </div>
                    </div>
                ))
            }
            </div>
        </div>)
    }
</div>
